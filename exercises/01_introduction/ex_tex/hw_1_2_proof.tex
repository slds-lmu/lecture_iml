\aufgabe{Correlation and fraction of explained variance}{

Show that the following relationship between the Pearson correlation coefficient and the coefficient of determination holds: 
\begin{align*}
	R^2 = \rho^2.
\end{align*}

This means that for two numerical one-dimensional variables, the degree of (linear) correlation between them is exactly equal to how well a simple linear model fits their relationship.

\textit{Recap:} 
\begin{align*}
	\rho &=  \frac{\sum_{i=1}^{n}{(x^{(i)}-\bar{x}) \cdot (y^{(i)}-\bar{y})}}{\sqrt{\sum_{i=1}^{n}(x^{(i)}-\bar{x})^2} \sqrt{\sum_{i=1}^{n}{(y^{(i)}-\bar{y})^2 }}}, \\
	R^2 &= 1-\frac{SSE_{LM}}{SSE_{c}},
\end{align*}
where 
\begin{align*}
	SSE_{LM} &= \sum_{i=1}^n (y^{(i)} - \hat{f}_{LM}(x^{(i)}))^2, \\
	SSE_{c} &= \sum_{i=1}^n (y^{(i)} - \bar{y})^2
\end{align*}
are the sum of squares due to regression and the total sum of squares, respectively.

}
