	\begin{table}[H]
		\centering

			1 & days\_since\_2011 & 1 873 827.22 & 1 733 044.28 \\ 
			2 & temp & 1 733 044.28 & 1 597 057.93 \\ 
			3 & days\_since\_2011 & 1 597 057.93 & 1 486 743.19 \\ 
			4 & temp & 1 486 743.19 & 1 379 888.98 \\ 
			5 & temp & 1 379 888.98 & 1 293 337.07 \\ 
			\hline
		\end{tabular}
	\end{table}
	
	Calculate the feature importance of the two features.
