\begin{enumerate}
    \item The covariate joint is preserved when sampling conditionally. Let $J$ be the set of resampled variables and $R$ be the remaining variables. Then it holds that
    \begin{align}
        \P(X) &= \P(X_J, X_R) &\\
        &= \P(X_J|X_R) \P(X_R) & \text{def. conditional distribution}\\
        &= \P(\pert{X}{J}{R}_J|X_R) \P(X_R) & \text{because } \P(\pert{X}{J}{R}_J|X_R) = \P(X_J|X_R) \\
        &= \P(\pert{X}{J}{R}_J, X_R) &
    \end{align}
    For CFI, $J = \{j\}$ where $j$ is the feature of interest. For SAGE value function $v(S)$ the features of interest are $J=S$.
\end{enumerate}
