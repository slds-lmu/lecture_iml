\begin{framev}[align=top]{Brief History of Interpretability}
\splitVTT[0.65]{
\begin{itemizeM}
\item 18th/19th century:\\Linear models (Gauss, Legendre, Quetelet)
\item Mid-20th century:\\Emergence of sensitivity analysis
\item Mid- to late 20th century:\\Rule-based ML, incl. decision rules and trees
\item Early 2000s:\\Interpretation tools for ensembles:
\begin{itemizeS}
\item variable importance for random forests
\item partial dependence for gradient boosting
\end{itemizeS}
\item Early 2010s:\\Explainability methods for deep learning
\item Late 2010s:\\IML / XAI as a distinct research area
\end{itemizeM}
