\begin{frame}{Resources and Software:}
\textbf{Individual Conditional Expectation (ICE)}
\begin{itemize}
    \item The paper: \citebutton{Goldstein et al. 2015}{https://www.tandfonline.com/doi/full/10.1080/10618600.2014.907095}
    \item Chapter in Interpretable Machine Learning book: \citebutton{Molnar 2025}{https://christophm.github.io/interpretable-ml-book/ice.html}
    \item R packages: \citebutton{iml}{https://github.com/giuseppec/iml}; \citebutton{pdp}{https://github.com/bgreenwell/pdp}; \citebutton{ICEbox}{https://github.com/kapelner/ICEbox}
    \item Python packages: \citebutton{PiML}{https://selfexplainml.github.io/PiML-Toolbox/_build/html/index.html}; \citebutton{sklearn}{https://scikit-learn.org/stable/modules/partial_dependence.html}
\end{itemize}

\medskip

\textbf{Partial Dependence (PD) plot}
\begin{itemize}
    \item Chapter in Interpretable Machine Learning book: \citebutton{Molnar 2025}{https://christophm.github.io/interpretable-ml-book/pdp.html}
    \item R packages: \citebutton{iml}{https://github.com/giuseppec/iml}; \citebutton{pdp}{https://github.com/bgreenwell/pdp}; \citebutton{DALEX}{https://github.com/ModelOriented/DALEX}
    \item Python packages: \citebutton{sklearn}{https://scikit-learn.org/stable/modules/partial_dependence.html}; \citebutton{effector}{https://github.com/givasile/effector}; \citebutton{PDPbox}{https://github.com/SauceCat/PDPbox}
\end{itemize}

\textbf{Accumulated Local Effect (ALE)}
\begin{itemize}
    \item The paper: \citebutton{Apley et al. 2020}{https://doi.org/10.1111/rssb.12377}
    \item Chapter in Interpretable Machine Learning book: \citebutton{Molnar 2025}{https://christophm.github.io/interpretable-ml-book/ale.html}
    \item R packages: \citebutton{ALEPlot}{https://cran.r-project.org/web/packages/ALEPlot/index.html}; \citebutton{iml}{https://github.com/giuseppec/iml}
    \item Python packages: \citebutton{effector}{https://github.com/givasile/effector}; \citebutton{PiML}{https://selfexplainml.github.io/PiML-Toolbox/_build/html/index.html}; \citebutton{ALEPython}{https://github.com/blent-ai/ALEPython}; \citebutton{Alibi}{https://docs.seldon.io/projects/alibi/en/stable/index.html}
\end{itemize}

\textbf{Regional effect plots (REPID)}
\begin{itemize}
    \item The paper: \citebutton{Herbinger et al. (2022)}{https://proceedings.mlr.press/v151/herbinger22a.html}
    \item Codebase: \citebutton{repid}{https://github.com/JuliaHerbinger/repid}
\end{itemize}
\end{frame}
